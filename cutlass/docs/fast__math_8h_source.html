158 CUTLASS_HOST_DEVICE
159 void find_divisor(unsigned int& mul, unsigned int& shr, unsigned int denom) {
160  if (denom == 1) {
161  mul = 0;
162  shr = 0;
163  } else {
164  unsigned int p = 31 + find_log2(denom);
165  unsigned m = unsigned(((1ull << p) + unsigned(denom) - 1) / unsigned(denom));
166 
167  mul = m;
168  shr = p - 32;
169  }
170 }
